const int buttonPin = 2;
int count = 0;

void setup() {
  pinMode(buttonPin, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop() {
  if (digitalRead(buttonPin) == LOW) {
    count++;

    Serial.print("Count = ");
    Serial.println(count);

    while (digitalRead(buttonPin) == LOW) {
      // Wait until button is released
    }

    delay(50);
  }
}
